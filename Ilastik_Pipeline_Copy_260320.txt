    "#Percentage of coverage quantification\n",
    "npy_files = out_folder.glob(\"*.npy\")\n",
    "\n",
    "npy_dict = {}\n",
    "\n",
    "for in_file in npy_files:\n",
    "    filename = in_file.stem\n",
    "    parts = filename.split(\"_\")\n",
    "    if len(parts) >= 6:\n",
    "        staining_method = parts[0]\n",
    "        genotype = parts[1]\n",
    "        gender = parts[2]\n",
    "        brain = parts[3]\n",
    "        slide = parts[4]\n",
    "        region = parts[5]\n",
    "\n",
    "        path = f\"{brain}*{region}_Simple Segmentation.npy\"\n",
    "        print(path)\n",
    "\n",
    "        data = np.load(str(in_file))\n",
    "        percentage_coverage = ((np.count_nonzero(data == 3) / (np.count_nonzero(data == 3) + np.count_nonzero(data == 2))) * 100)\n",
    "        unique_key = f\"{staining_method}_{genotype}_{gender}_{brain}_{region}\"\n",
    "\n",
    "        if unique_key not in npy_dict:\n",
    "            npy_dict[unique_key] = {'slide': [], 'a': []}\n",
    "\n",
    "        npy_dict[unique_key]['slide'].append(slide)\n",
    "        npy_dict[unique_key]['a'].append(percentage_coverage)\n",
    "\n",
    "for key, value in npy_dict.items():\n",
    "    slide = value['slide']\n",
    "    a = value['a']\n",
    "    total_percentage_coverage = sum(a) / len(a)\n",
    "    a.append(total_percentage_coverage)\n",
    "    slide.append(\"AVERAGE\")\n",
    "    b = key.replace(\"*\", \"_\")\n",
    "\n",
    "    e = zip(slide, a)\n",
    "\n",
    "    df = pd.DataFrame(e, columns=[\"Slide\", \"Percentage Coverage\"])\n",
    "\n",
    "    output_filename = f\"{b}_Percentage_Of_Coverage.xlsx\"\n",
    "    output_path = out_folder / output_filename\n",
    "\n",
    "    writer = pd.ExcelWriter(output_path, engine='xlsxwriter')           \n",
    "    df.to_excel(writer, sheet_name='Percentage of Coverage', index=False)\n",
    "    writer.close()"

    "#Quantification of plaque classification\n",
    "csv_files = out_folder.glob(\"*.csv\")\n",
    "\n",
    "csv_dict = {}\n",
    "\n",
    "for in_file in csv_files:\n",
    "    filename = in_file.stem\n",
    "    parts = filename.split(\"_\")\n",
    "    \n",
    "    if len(parts) >= 6:\n",
    "        staining_method = parts[0]\n",
    "        genotype = parts[1]\n",
    "        gender = parts[2]\n",
    "        brain = parts[3]\n",
    "        slide = parts[4]\n",
    "        region = parts[5]\n",
    "        \n",
    "        unique_key = f\"{brain}_{region}\"\n",
    "        \n",
    "        if unique_key not in csv_dict:\n",
    "            csv_dict[unique_key] = {'data': []}\n",
    "        \n",
    "        read = pd.read_csv(in_file)\n",
    "        csv_dict[unique_key]['data'].append(read)\n",
    "\n",
    "for key, value in csv_dict.items():\n",
    "    data = pd.concat(value['data'])\n",
    "    c = data['Predicted Class'].value_counts(normalize=True) * 100\n",
    "    \n",
    "    class_dist_row = pd.DataFrame([c], columns=c.index)\n",
    "    result = pd.concat([class_dist_row, data], ignore_index=True)\n",
    "    \n",
    "    output_filename = f\"{key}_SML.xlsx\"\n",
    "    output_path = out_folder / output_filename\n",
    "    \n",
    "    writer = pd.ExcelWriter(output_path, engine='xlsxwriter')\n",
    "    result.to_excel(writer, sheet_name='SML', index=False)\n",
    "    writer.close()"
